```python
#도시별 평균나이구하기 [groupby 없이]
city_age ={}
city_count = {}

for index, row in data.iterrows():
    city = row['City']
    age = row['Age']
    
    if city in city_age:
        city_age[city] +=age
        city_count[city] +=1
    else:
        city_age[city] =age
        city_count[city] =1
        
average = {city : city_age[city] / city_count[city] for city in city_age}

for city , age in average.items():
    print(f'{city} : {age:.2f}')
```
